<program>        => <form>* EOFToken

<form>           => <definition>
                  |  <expression>

<definition>     => ( define <symbol> <expression> )
                  |  ( define ( <symbol> <def-formals> ) <expression> )
                  |  ( begin <definition>* )



<expression>     => <symbol>
                  |  <literal>
                  |  <lambda>
                  |  <conditional>
                  |  <application>


<lambda>         => ( lambda ( <symbol>* ) <body> )
<conditional>    => ( if <test> <consequent> <alternative> )
<application>    => ( <operator> <operand>* )

<symbol>         => variable_name

<literal>        => <quotation>
                  |  <integer>
                  |  <boolean>

<def-formals>    => <symbol>*
// <expression>* <expression> == <expression>+ BUT the last
// expression is necessary
<body>           => <definition>* <expression>* <expression>

<test>           => <expression>
<consequent>     => <expression>
<alternative>    => <expression>
                  |  <>

<operator>       => <expression>
<operand>        => <expression>

<quotation>      => ( quote <expression> )
                  |  '<expression>

<integer>        => int_value
<boolean>        => #f | #t
